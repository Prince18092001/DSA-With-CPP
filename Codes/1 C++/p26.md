Q. Add Two String (Add number in string format)
ans:- In this question we add two number that are in string formate ( large number string me stored hota ah because any data type has fixed length and arise where overflow condition) 
   mean, we given two number in string format aur isa hama add karka result string format me dena ha

steps:
it do as similar to add two number by taking one by one from last digit of both number then add last two digit and carry generate and so on.
 Example: 
      carry       1 1  
      num1      2 4 6 8
      num2        2 4 6
      add       2 7 1 4

code:-

// ADD Strings
// when n1 >= n2
#include<iostream>
#include <algorithm>
using namespace std;
int main(){
     string num1, num2, ans;
     cout<<"Enter the First number (n1>=n2): ";
     getline(cin,num1);
     cout<<"Enter the Second number: ";
     getline(cin,num2);
     int index1 = num1.size()-1, index2 = num2.size()-1, carry = 0, sum = 0;
    
     while(index2>=0){
        sum = (num1[index1]-'0') + (num2[index2]-'0') + carry;
        carry = sum/10;
        char c = sum%10 + '0';   // integer + ASCII value of 0   (it convert integer into char)
        ans += c;
        index1--, index2--;
     }

     while(index1>=0){
        sum = (num1[index1]-'0')  + carry;
        carry = sum/10;
        char c = sum%10 + '0';
        ans += c;
        index1--;
     }

     if(carry){
        ans += '1';   // 2 number addation  me always carry 1 he ati ah
     }
     reverse(ans.begin(), ans.end());
    
     cout<<ans;
}

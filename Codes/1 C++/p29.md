Q. Longest Substring Without Repeating Characters.
ans:- In this question we find longest substring without repeating character(taking continuous string (not change order / select random)).

steps:-
       Create vector in which store 1/0 1 mean character present 0 mean not present 
       Iterate over string and fill the vector:
                 1. if vector cell is zero then count 1.
                 2. store length in length variable.
                 3. if vector cell is 1(mean value repeat) then empty cell from start to   previous char that are currently repeat.
                 4. repeat this steps.
                 5. return length.

code:-

// Longest Substring Without Repeating Characters
//It is solve by Sliding Window Protocol
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
    string s = "abcdecbeadf"; //ababc
    vector<bool>count(256,0);
    int first=0, second=0, len=0;
    while(second<s.size()){
        //if char repeat
        while(count[s[second]]){  // It check that character already present in vector or not if present then remove by initilize count 0 from start to previous char that currently repeat
           count[s[first]] = 0;
           first++;
        }
        //simple count
        count[s[second]] = 1;
        len = max(len,second-first+1);
        second++;
    }
    cout<<len;
}

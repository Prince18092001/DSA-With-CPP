Q. Factorials of Large number.
ans:- If we try to find factorial of large number by using int data type or other like long int then in which value overflow because its size has fixed(small) and values are very large. So, to solve this problem we use string because we can store large value in string but it is diffcult(where every time value convert from string to integer(like this '8'-'0') So, we use vector/array to solve this problem its is better approach than string).

note:- vector size increase during runtime by using push_back() function.
push_back() : It insert at last but at insert time if vector cell are full then it create new array of double size.

code:-

// Factorial of Large Number
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
    int n;
    cout<<"Enter a number: ";
    cin>>n;

    vector<int>ans(1,1);

    while(n>1){
        int carry = 0, result, size = ans.size();

        for(int i=0; i<size; i++){   // start from 0 position of vector because after become more value then it store to increse size of vector
            result = ans[i]*n + carry;
            ans[i] = result%10;
            carry = result/10;
        }
        while(carry){
            ans.push_back(carry%10); // if cell of vector become full then it create double size new vector to insert element
            carry /= 10;            // if carry become two digit then insert in vector different cell
        }
       n--;
    }
    
    reverse(ans.begin(),ans.end());
    for(int i=0; i<ans.size(); i++){
          cout<<ans[i];
    }
}
